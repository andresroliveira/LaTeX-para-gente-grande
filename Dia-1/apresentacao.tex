% Bloco 2 - Ecossistema LaTeX
\begin{frame}{LaTeX não é um programa único}
    O que chamamos de ``LaTeX'' é na verdade um conjunto de ferramentas:
    \begin{itemize}
        \item \textbf{Distribuição}: O conjunto completo (TeX Live, MiKTeX).
        \item \textbf{Editor}: Onde você digita (TeXstudio, VS Code, Vim).
        \item \textbf{Motor}: O programa que compila (\texttt{pdflatex}).
        \item \textbf{Classe}: O esqueleto do documento (\texttt{article}, \texttt{book}).
        \item \textbf{Pacote}: Funcionalidades extras (\texttt{amsmath}, \texttt{tikz}).
    \end{itemize}
\end{frame}

\begin{frame}{Distribuição: TeX Live e MiKTeX}
    \begin{itemize}
        \item \textbf{TeX Live}: Padrão no Linux/Mac, multiplataforma, muito estável.
        \item \textbf{MiKTeX}: Muito comum no Windows, instala pacotes sob demanda.
    \end{itemize}
    \vfill
    \begin{block}{Dica técnica}
        Sempre prefira a instalação completa (Full) para evitar problemas de dependências durante o trabalho offline.
    \end{block}
\end{frame}

\begin{frame}{Editor: Onde escrevemos}
    O editor não é o compilador. Ele apenas chama o motor.
    \begin{itemize}
        \item \textbf{TeXstudio}: Especializado, possui assistentes para tabelas.
        \item \textbf{VS Code}: Moderno, exige extensões (\textit{LaTeX Workshop}).
        \item \textbf{Overleaf}: Editor na nuvem (baseado em TeX Live).
    \end{itemize}
\end{frame}

\begin{frame}{Motor: Quem processa}
    O binário que lê o seu \texttt{.tex} e gera o PDF.
    \begin{itemize}
        \item \texttt{pdflatex}: O nosso foco.
        \item \texttt{xelatex}: O favorito para fontes do sistema.
        \item \texttt{lualatex}: O mais moderno e extensível.
    \end{itemize}
\end{frame}

\begin{frame}{Classe: Estrutura do documento}
    Define os comandos de alto nível (\texttt{\textbackslash chapter}, \texttt{\textbackslash section}).
    \begin{itemize}
        \item \texttt{article}: Para artigos curtos.
        \item \texttt{book}: Para livros (possui \texttt{\textbackslash frontmatter}, \texttt{\textbackslash mainmatter}).
        \item \texttt{beamer}: Para apresentações (como esta!).
    \end{itemize}
\end{frame}

\begin{frame}{Pacote: Extensão de funcionalidade}
    Carregados no preâmbulo com \texttt{\textbackslash usepackage}.
    \begin{itemize}
        \item \texttt{amsmath}: Matemática profissional.
        \item \texttt{graphicx}: Inclusão de imagens.
        \item \texttt{hyperref}: Links e sumário clicável.
    \end{itemize}
\end{frame}

\begin{frame}{Confusões Comuns}
    \begin{itemize}
        \item ``O LaTeX não instalou no meu PC'': Provavelmente você instalou o editor, mas não a distribuição.
        \item ``A fonte X não funciona'': Pode ser incompatibilidade entre o pacote da fonte e o motor de compilação.
        \item ``O sumário sumiu'': Falta de compilações subsequentes.
    \end{itemize}
\end{frame}

% Bloco 3 - Fluxo de compilação
\begin{frame}{Fluxo Básico de Compilação}
    \begin{center}
        \texttt{.tex} $\to$ \textbf{Motor} $\to$ \texttt{.pdf}
    \end{center}
    \vfill
    O motor lê o texto, aplica as regras da classe e dos pacotes, e gera o documento final.
\end{frame}

\begin{frame}{Arquivos Auxiliares}
    A compilação gera subprodutos necessários:
    \begin{itemize}
        \item \texttt{.aux}: O mais importante. Guarda a memória do documento.
        \item \texttt{.log}: O diário da compilação.
        \item \texttt{.toc}: Sumário (\textit{Table of Contents}).
        \item \texttt{.bbl}: Bibliografia formatada.
    \end{itemize}
\end{frame}

\begin{frame}{O arquivo .aux: Memória entre compilações}
    \begin{itemize}
        \item Na 1ª vez: O motor vê um \texttt{\textbackslash label}, anota no \texttt{.aux}.
        \item Na 2ª vez: O motor lê o \texttt{.aux} e resolve o \texttt{\textbackslash ref}.
    \end{itemize}
    \vfill
    \begin{alertblock}{Importante}
        Se você deletar o \texttt{.aux}, o próximo PDF terá ``??'' no lugar das referências.
    \end{alertblock}
\end{frame}

\begin{frame}{O arquivo .log: Registro e Diagnóstico}
    \begin{itemize}
        \item Contém avisos de pacotes obsoletos.
        \item Lista as fontes que foram realmente embutidas no PDF.
        \item É o primeiro lugar a olhar quando a compilação falha.
    \end{itemize}
\end{frame}

\begin{frame}{Por que compilar mais de uma vez?}
    \begin{enumerate}
        \item \textbf{1ª Passagem}: Coleta dados estruturais (títulos, labels).
        \item \textbf{2ª Passagem}: Constrói o sumário e resolve referências.
        \item \textbf{3ª Passagem}: Garante que o layout final (com sumário) não mudou as numerações de página.
    \end{enumerate}
\end{frame}

% Bloco 4 - Motores de compilação
\begin{frame}{pdflatex}
    \begin{itemize}
        \item Rápido e extremamente estável.
        \item Modelo de fontes Type 1 (pacotes de fontes).
        \item Excelente suporte ao \texttt{microtype}.
    \end{itemize}
\end{frame}

\begin{frame}{xelatex e lualatex}
    \begin{itemize}
        \item \textbf{XeLaTeX}: Suporte a Unicode e fontes do sistema (\texttt{.otf}, \texttt{.ttf}).
        \item \textbf{LuaLaTeX}: Moderno, permite programar o comportamento do TeX usando Lua.
    \end{itemize}
\end{frame}

\begin{frame}{Por que o curso usa pdfLaTeX?}
    \begin{itemize}
        \item Compatibilidade garantida em periódicos científicos.
        \item Estabilidade absoluta para documentos técnicos.
        \item Não exige instalação de fontes específicas no sistema do usuário.
    \end{itemize}
\end{frame}

% Bloco 5 - Fontes no pdfLaTeX
\begin{frame}{A fonte padrão: Computer Modern}
    \begin{itemize}
        \item Criada por Donald Knuth.
        \item Extremamente técnica e precisa para matemática.
        \item Visual clássico do TeX dos anos 80.
    \end{itemize}
\end{frame}

\begin{frame}{UTF-8 e Codificação de Entrada}
    \begin{itemize}
        \item Como o seu computador salva o arquivo \texttt{.tex}.
        \item Use sempre \textbf{UTF-8}.
        \item \texttt{\textbackslash usepackage[utf8]\{inputenc\}} (hoje é o padrão).
    \end{itemize}
\end{frame}

\begin{frame}{fontenc e T1}
    \begin{itemize}
        \item \textbf{T1}: Essencial para línguas latinas.
        \item Permite hifenação correta de palavras acentuadas.
        \item Permite busca de caracteres como ``ç'' e ``á'' no PDF.
    \end{itemize}
    \vfill
    \texttt{\textbackslash usepackage[T1]\{fontenc\}}
\end{frame}

\begin{frame}{Pacotes de Fontes}
    Não usamos o arquivo da fonte diretamente, mas um pacote que o encapsula.
    \begin{itemize}
        \item \texttt{ebgaramond}: Elegância clássica.
        \item \texttt{lmodern}: Versão melhorada da padrão.
        \item \texttt{libertine}: Ótima para textos longos.
    \end{itemize}
\end{frame}

\begin{frame}{Microtype: O toque final}
    Ajustes finos invisíveis.
    \begin{itemize}
        \item \textbf{Protrusão}: Alinhamento ótico das margens.
        \item \textbf{Expansão}: Ajuste na largura dos glifos para evitar buracos no texto.
    \end{itemize}
\end{frame}

\begin{frame}{fontspec: Apenas para XeLaTeX e LuaLaTeX}
    \begin{itemize}
        \item Seleciona fontes do sistema pelo nome.
        \item \textbf{Não funciona com pdflatex}.
    \end{itemize}
\end{frame}

\begin{frame}{Exemplo comparativo: fontes locais}
    \begin{block}{XeLaTeX / LuaLaTeX apenas}
        \texttt{\textbackslash setmainfont\{MinhaFonte.otf\}}
    \end{block}
    \vfill
    \textit{Este recurso não é usado na configuração principal deste projeto.}
\end{frame}

% Bloco 6 - Automação
\begin{frame}{O problema da compilação manual repetida}
    Rodar manualmente:
    \begin{enumerate}
        \item \texttt{pdflatex principal.tex}
        \item \texttt{bibtex principal.aux}
        \item \texttt{pdflatex principal.tex}
        \item \texttt{pdflatex principal.tex}
    \end{enumerate}
    É cansativo e propenso a esquecimentos.
\end{frame}

\begin{frame}{latexmk: Automação Total}
    \begin{itemize}
        \item Detecta se o arquivo mudou.
        \item Roda o compilador o número de vezes necessário.
        \item Limpa arquivos temporários se solicitado.
    \end{itemize}
\end{frame}

\begin{frame}{Makefile: Padrão de Projeto}
    Encapsula o \texttt{latexmk} em um comando simples.
    \vfill
    No nosso projeto: \texttt{make}
\end{frame}

% Bloco 7 - Erros e Diagnóstico
\begin{frame}{Estratégia de Diagnóstico}
    \begin{alertblock}{Regra de Ouro}
        Leia sempre a \textbf{primeira} mensagem de erro.
    \end{alertblock}
    \vfill
    Os erros seguintes costumam ser apenas confusão do compilador tentando se recuperar do primeiro.
\end{frame}

\begin{frame}{Undefined control sequence}
    \begin{itemize}
        \item Você digitou o comando errado (ex: \texttt{\textbackslash secton}).
        \item Você esqueceu de carregar o pacote que define o comando.
    \end{itemize}
\end{frame}

\begin{frame}{Chaves e ambientes não fechados}
    \begin{itemize}
        \item \texttt{Runaway argument?}: Falta fechar um \texttt{\}}.
        \item \texttt{\textbackslash begin\{itemize\}} sem o respectivo \texttt{\textbackslash end\{itemize\}}.
    \end{itemize}
\end{frame}

\begin{frame}{Overfull e Underfull \textbackslash hbox}
    \begin{itemize}
        \item \textbf{Overfull}: O texto ultrapassou a margem (estouro).
        \item \textbf{Underfull}: O texto ficou muito espalhado (espaços feios).
    \end{itemize}
\end{frame}
