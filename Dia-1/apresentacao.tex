% BLOCO 2 - ECOSSISTEMA
\begin{frame}{LaTeX não é um programa único}
    \begin{block}{O Ecossistema TeX}
        LaTeX é um conjunto de macros executado por um compilador TeX e integrado a outras ferramentas.
    \end{block}
    \vfill
    \textbf{Principais componentes:}
    \begin{enumerate}
        \item Distribuição (O repositório).
        \item Compilador ou motor de compilação.
        \item Editor (A interface).
        \item Classe e pacote.
    \end{enumerate}
\end{frame}

\begin{frame}{Distribuições: O Repositório de Ferramentas}
    \begin{itemize}
        \item \textbf{TeX Live}: distribuição multiplataforma (Linux, macOS, Windows).
        \item \textbf{MiKTeX}: Popular no Windows, instala pacotes sob demanda.
        \item \textbf{MacTeX}: empacotamento do TeX Live para macOS.
    \end{itemize}
    \vfill
    \begin{block}{Observação técnica}
        Ambientes online como o \textbf{Overleaf} utilizam distribuições TeX Live nos bastidores.
    \end{block}
\end{frame}

\begin{frame}{Editores e compiladores}
    O editor \textbf{não é} o compilador. Ele apenas chama o programa que faz o trabalho.
    \begin{itemize}
        \item \textbf{TeXstudio}: IDE completa com visualizadores de log e PDF integrados.
        \item \textbf{VS Code + LaTeX Workshop}: editor extensível com integração ao compilador.
        \item \textbf{Overleaf}: editor na nuvem com compilação integrada.
        \item \textbf{Emacs + AUCTeX}: ambiente local com automação e comandos especializados.
    \end{itemize}
\end{frame}

\begin{frame}{Compilador: quem processa}
    No ecossistema TeX, programas como \texttt{pdflatex}, \texttt{xelatex} e \texttt{lualatex} também são chamados de motores ou \textit{engines}.
    \begin{itemize}
        \item \textbf{\texttt{pdflatex}}: Estável, rápido, foco em fontes tradicionais.
        \item \textbf{\texttt{xelatex}}: Unicode nativo, usa fontes do sistema (.ttf, .otf).
        \item \textbf{\texttt{lualatex}}: Extensível via scripts Lua, suporte a fontes modernas.
    \end{itemize}
\end{frame}

\begin{frame}{Classe e Pacote}
    \begin{itemize}
        \item \textbf{Classe (\texttt{.cls})}: O esqueleto.
              \begin{itemize}
                  \item \texttt{article}, \texttt{book}, \texttt{beamer}, \texttt{abntex2}.
              \end{itemize}
        \item \textbf{Pacote (\texttt{.sty})}: Funcionalidades específicas.
              \begin{itemize}
                  \item \texttt{amsmath} (matemática), \texttt{graphicx} (imagens), \texttt{tikz} (desenho).
              \end{itemize}
    \end{itemize}
\end{frame}

% BLOCO 3 - FLUXO DE COMPILAÇÃO
\begin{frame}[fragile]{Anatomia da Compilação}
    \begin{center}
        \texttt{arquivo.tex} $\to$ \textbf{Compilador} $\to$ \texttt{arquivo.pdf}
    \end{center}
\begin{lstlisting}
pdflatex arquivo.tex
ls arquivo.*
\end{lstlisting}
    \textbf{Arquivos auxiliares:}
    \begin{itemize}
        \item \textbf{\texttt{.aux}}: Labels, citações e contadores entre passagens.
        \item \textbf{\texttt{.log}}: Registro de erros, avisos, pacotes e arquivos carregados.
        \item \textbf{\texttt{.toc}}: Estrutura usada para gerar o sumário.
        \item \textbf{\texttt{.bbl}}: A bibliografia formatada pelo BibTeX.
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{O Ciclo das Referências}
    Por que compilar mais de uma vez?
\begin{lstlisting}[language={[latex]TeX}]
Veja a Secao~\ref{sec:intro}.

\section{Intro}\label{sec:intro}
\end{lstlisting}
    \textbf{1ª passagem:} escreve o label no \texttt{.aux}.\\
    \textbf{2ª passagem:} lê o \texttt{.aux} e resolve o número.
    \begin{alertblock}{Aviso clássico}
        \texttt{LaTeX Warning: Label(s) may have changed. Rerun to get cross-references right.}
    \end{alertblock}
\end{frame}

% BLOCO 5 - MOTORES E FONTES
\begin{frame}{pdflatex, xelatex e lualatex}
    \begin{itemize}
        \item \textbf{pdflatex}: usa o modelo tradicional de fontes do TeX.
        \item \textbf{xelatex/lualatex}: podem usar fontes do sistema com \texttt{fontspec}.
        \item \textbf{Decisão do projeto}: o livro e os slides compilam com \texttt{pdflatex}.
    \end{itemize}
    \vfill
    \begin{block}{Motivo prático}
        Reduz dependências de fontes instaladas no sistema operacional de cada participante.
    \end{block}
\end{frame}

\begin{frame}[fragile]{Fontes no pdfLaTeX}
    No \texttt{pdfLaTeX}, a configuração básica envolve três elementos:
    \begin{itemize}
        \item \textbf{UTF-8}: Codificação de entrada.
        \item \textbf{T1}: Codificação de saída (glifos acentuados únicos).
        \item \textbf{Pacote}: A família tipográfica.
    \end{itemize}
    \vfill
\begin{lstlisting}[language={[latex]TeX}]
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}
\usepackage{ebgaramond}
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Fontes por pacote}
    No \texttt{pdfLaTeX}, a troca de família tipográfica normalmente passa por pacotes.
\begin{lstlisting}[language={[latex]TeX}]
\usepackage[T1]{fontenc}
\usepackage{newpxtext}
\usepackage{newpxmath}
\end{lstlisting}
    \vfill
    \textbf{Consequência:} o documento depende da distribuição TeX, não das fontes instaladas pelo sistema operacional.
\end{frame}

\begin{frame}{Por que usar T1?}
    \begin{itemize}
        \item \textbf{Sem T1}: ``ç'' vira ``c'' + ``acento'', quebra a hifenação e a busca no PDF.
        \item \textbf{Com T1}: O PDF contém o glifo real de ``ç''.
    \end{itemize}
    \vfill
    \textit{Teste no PDF final: copie uma palavra com acento e verifique se o texto extraído preserva os caracteres.}
\end{frame}

\begin{frame}[fragile]{Microtipografia}
    O pacote \texttt{microtype} faz ajustes finos na composição de parágrafos.
\begin{lstlisting}[language={[latex]TeX}]
\usepackage{microtype}
\end{lstlisting}
    \begin{itemize}
        \item \textbf{Protrusão}: pontuação pode avançar levemente na margem.
        \item \textbf{Expansão}: pequenas variações na largura dos caracteres.
        \item \textbf{Limite}: não resolve URLs longas, tabelas largas ou texto mal estruturado.
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Fontes do sistema}
    Em \texttt{XeLaTeX} ou \texttt{LuaLaTeX}, \texttt{fontspec} pode carregar fontes instaladas no sistema:
\begin{lstlisting}[language={[latex]TeX}]
\usepackage{fontspec}
\setmainfont{TeX Gyre Pagella}
\setsansfont{TeX Gyre Heros}
\setmonofont{DejaVu Sans Mono}
\end{lstlisting}
    \vfill
    \begin{alertblock}{Importante}
        Este código não pertence ao preâmbulo real do projeto, que usa \texttt{pdflatex}.
    \end{alertblock}
\end{frame}

\begin{frame}[fragile]{Arquivos locais .ttf ou .otf}
    Também é possível distribuir os arquivos de fonte junto com o projeto:
\begin{lstlisting}[language={[latex]TeX}]
\usepackage{fontspec}
\setmainfont[
  Path = fontes/,
  UprightFont = MinhaFonte-Regular.ttf,
  BoldFont = MinhaFonte-Bold.ttf
]{MinhaFonte}
\end{lstlisting}
    \vfill
    \textbf{Trade-off:} melhora o controle sobre a fonte usada, mas exige cuidar de caminhos e licença.
\end{frame}

% BLOCO 6 - AUTOMAÇÃO
\begin{frame}[fragile]{Compilação manual}
    Para um projeto com bibliografia:
\begin{lstlisting}
pdflatex doc.tex
bibtex doc
pdflatex doc.tex
pdflatex doc.tex
\end{lstlisting}
    \vfill
    \textbf{Risco:} esquecer uma etapa deixa referências, citações ou sumário desatualizados.
\end{frame}

\begin{frame}[fragile]{Ferramentas de Automação}
\begin{lstlisting}
latexmk -pdf doc.tex
make
\end{lstlisting}
    \begin{itemize}
        \item \textbf{\texttt{latexmk}}: monitora dependências e recompila o necessário.
        \item \textbf{Makefile}: padroniza o build do projeto.
        \item \textbf{Editor/Overleaf}: chamam comandos semelhantes por trás da interface.
    \end{itemize}
\end{frame}

% BLOCO 7 - ERROS E DIAGNÓSTICO
\begin{frame}{Estratégia de diagnóstico}
    \begin{alertblock}{Ponto de partida}
        Sempre ignore os últimos erros e leia o \textbf{primeiro} erro relevante.
    \end{alertblock}
    \vfill
    \textbf{Sequência prática:}
    \begin{itemize}
        \item Localize a primeira linha iniciada por \texttt{!}.
        \item Identifique o arquivo e o número da linha.
        \item Leia algumas linhas antes do ponto indicado.
        \item Diferencie erro, aviso e sugestão de recompilação.
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Undefined control sequence}
    \textbf{Causa:} O TeX viu um comando que ele não conhece.
\begin{lstlisting}
! Undefined control sequence.
l.12 \seciton
             {Introducao}
\end{lstlisting}
    \textbf{Leitura:} o comando na linha 12 está escrito de forma incorreta ou depende de pacote não carregado.
\end{frame}

\begin{frame}[fragile]{Missing \$ inserted}
    \textbf{Causa:} Uso de caracteres reservados fora do ambiente matemático.
\begin{lstlisting}
O arquivo dados_2025.csv foi importado.
\end{lstlisting}
    \textbf{Problema:} \texttt{\_} tem significado especial.\\
    \textbf{Correção:}
\begin{lstlisting}
O arquivo dados\_2025.csv foi importado.
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Overfull \textbackslash hbox}
    \textbf{Sintoma:} Texto saindo da margem no PDF.
\begin{lstlisting}
Overfull \hbox (12.3pt too wide)
in paragraph at lines 18--20
\end{lstlisting}
    \textbf{Causas comuns:} palavra longa, URL, comando que não quebra linha ou trecho que precisa ser reescrito.
\end{frame}

    \textbf{Tarefa:}

        \item No arquivo \texttt{ola.tex}, digite: \texttt{\textbackslash errinho\{Texto\}}.

    \textbf{Verificação:} Apague o erro, recompile e veja se o arquivo .aux foi atualizado.
